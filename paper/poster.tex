% COLUMN 1: Problem + Architecture
\begin{column}{\colwidth}
\begin{block}{Problem}
Multi-hop QA needs multiple reasoning + retrieval steps. Standard RAG retrieves once---single-hop only.
\vspace{6pt}
\begin{center}
\includegraphics[width=0.9\colwidth]{figs/architecture.png}
\end{center}
\end{block}

\vspace{\sep}

\begin{block}{Design Principles}
\begin{itemize}
    \item \textbf{Fact-Free Synapses} --- 0\% facts in planner weights, 100\% routing
    \item \textbf{Entropic Collapse} --- convergence via entropy plateau
    \item \textbf{Offline-First} --- zero API keys, any hardware
\end{itemize}
\end{block}
\end{column}

% COLUMN 2: Method
\begin{column}{\colwidth}
\begin{block}{Gravitational Core (Planner)}
Holds zero factual knowledge. State: goal\_vector, spiral\_memory, hop\_history, confidence/entropy. Output: XML \texttt{<step>}, \texttt{<final\_answer>}, \texttt{<stop\_search>}.
\end{block}

\vspace{\sep}

\begin{block}{Centrifugal Ingestor (Executor)}
Fixed chained pipeline (no LLM tool selection):
\begin{enumerate}
    \item keyword\_search $\rightarrow$ lexical match
    \item chunk\_read $\rightarrow$ full text + neighbors
    \item LLM $\rightarrow$ condensed \texttt{<fact>}
\end{enumerate}
\end{block}

\vspace{\sep}

\begin{block}{Cyclic Spiral Flow}
Planner $\rightarrow$ \texttt{<step>} $\rightarrow$ Executor $\rightarrow$ \texttt{<fact>} $\rightarrow$ ... $\rightarrow$ convergence.

\vspace{4pt}
Termination: confidence $\ge$ 0.85, entropy stall $\times$3, or max hops (15).
\end{block}
\end{column}

\begin{block}{Key Findings}
\begin{itemize}
    \item Architecture validated: 72\% Contains on 7B CPU
    \item Model size critical: $\le$0.5B fails XML contract
    \item Fixed pipeline beats LLM tool selection
    \item EM 40\%+ expected with GPT-4o-mini (Phase 2)
\end{itemize}
\end{block}
